
## 快速排序-栈
**“显式栈（非递归）实现”**：Lomuto、Hoare、挖坑法、三路快排，以及随机 pivot、以及“小数组切插入排序”的工程优化版。

> 约定：都 **原地排序**（in-place），函数返回 `a` 方便链式使用。



### 0) 公共：插入排序（用于优化版）

```python
from __future__ import annotations
from typing import List, Tuple
import random

def insertion_sort_range(a: List[int], lo: int, hi: int) -> None:
    # 对闭区间 [lo, hi] 做原地插入排序（小区间时常比快排更快）
    for i in range(lo + 1, hi + 1):
        key = a[i]  # 当前待插入元素
        j = i - 1   # 从已排序区间末尾向前寻找插入位置
        # 将所有比 key 大的元素右移一位
        while j >= lo and a[j] > key:
            a[j + 1] = a[j]
            j -= 1
        # key 放到最终位置
        a[j + 1] = key
```

### 1) Lomuto 分区 + 栈快排

>partition（pivot 在 hi）

```python
def partition_lomuto(a: List[int], lo: int, hi: int) -> int:
    pivot = a[hi]   # 选最后一个元素作为 pivot
    i = lo - 1      # i 指向“< pivot 区域”的最后一个位置
    for j in range(lo, hi):
        # 如果当前元素比 pivot 小，就扩展左侧小于区并交换到前面
        if a[j] < pivot:
            i += 1
            a[i], a[j] = a[j], a[i]
    # 扫描完成后，把 pivot 放到小于区后面（它的最终有序位置）
    a[i + 1], a[hi] = a[hi], a[i + 1]
    return i + 1
```

>栈版 quicksort（带“先压大段、后处理小段”减少栈深）

```python
def quicksort_lomuto_stack(a: List[int]) -> List[int]:
    n = len(a)
    # 长度 0/1 的数组天然有序
    if n < 2:
        return a

    # 栈里存待排序区间 (lo, hi)
    stack: List[Tuple[int, int]] = [(0, n - 1)]
    while stack:
        # 每次取出一个区间做 partition
        lo, hi = stack.pop()
        if lo >= hi:
            continue

        # 用 Lomuto 分区，p 是 pivot 最终位置
        p = partition_lomuto(a, lo, hi)
        left = (lo, p - 1)
        right = (p + 1, hi)

        # 先压大的，后处理小的（让栈更浅）
        if (left[1] - left[0]) > (right[1] - right[0]):
            if left[0] < left[1]: stack.append(left)
            if right[0] < right[1]: stack.append(right)
        else:
            if right[0] < right[1]: stack.append(right)
            if left[0] < left[1]: stack.append(left)

    # 与递归版一致，排序在原数组上完成
    return a
```

### 2) Hoare 分区 + 栈快排

>partition（返回分割点 j；递归区间是 [lo..j] 和 [j+1..hi]）

```python
def partition_hoare(a: List[int], lo: int, hi: int) -> int:
    pivot = a[lo]   # 常见实现：取区间第一个元素为 pivot
    i = lo - 1      # i 从左边界外开始向右走
    j = hi + 1      # j 从右边界外开始向左走
    while True:
        # 找到左侧第一个 >= pivot 的位置
        i += 1
        while a[i] < pivot:
            i += 1
        # 找到右侧第一个 <= pivot 的位置
        j -= 1
        while a[j] > pivot:
            j -= 1
        # 指针交错时，j 即分割点
        if i >= j:
            return j
        # 把左侧大元素和右侧小元素交换
        a[i], a[j] = a[j], a[i]
```

>栈版 quicksort

```python
def quicksort_hoare_stack(a: List[int]) -> List[int]:
    n = len(a)
    # 长度 0/1 直接返回
    if n < 2:
        return a

    # 栈保存待处理区间
    stack: List[Tuple[int, int]] = [(0, n - 1)]
    while stack:
        lo, hi = stack.pop()
        if lo >= hi:
            continue

        # Hoare 分区返回分割点 p，子区间是 [lo, p] 和 [p+1, hi]
        p = partition_hoare(a, lo, hi)
        left = (lo, p)
        right = (p + 1, hi)

        # 先压较大区间，降低最坏情况下的栈深
        if (left[1] - left[0]) > (right[1] - right[0]):
            if left[0] < left[1]: stack.append(left)
            if right[0] < right[1]: stack.append(right)
        else:
            if right[0] < right[1]: stack.append(right)
            if left[0] < left[1]: stack.append(left)

    return a
```

### 3) 挖坑法 partition + 栈快排

>partition（pivot 在 lo，最后返回 pivot 落点）

```python
def partition_hole(a: List[int], lo: int, hi: int) -> int:
    pivot = a[lo]   # 暂存 pivot，a[lo] 位置成为“坑”
    i, j = lo, hi   # i 从左向右，j 从右向左
    while i < j:
        # 从右侧找第一个 < pivot 的元素，填到左边坑位
        while i < j and a[j] >= pivot:
            j -= 1
        a[i] = a[j]

        # 从左侧找第一个 > pivot 的元素，填到右边坑位
        while i < j and a[i] <= pivot:
            i += 1
        a[j] = a[i]

    # i == j 时，把 pivot 填回最终坑位
    a[i] = pivot
    return i
```

>栈版 quicksort

```python
def quicksort_hole_stack(a: List[int]) -> List[int]:
    n = len(a)
    # 空数组或单元素无需排序
    if n < 2:
        return a

    # 栈里保存闭区间
    stack: List[Tuple[int, int]] = [(0, n - 1)]
    while stack:
        lo, hi = stack.pop()
        if lo >= hi:
            continue

        # 挖坑法分区，返回 pivot 落点 p
        p = partition_hole(a, lo, hi)
        left = (lo, p - 1)
        right = (p + 1, hi)

        # 压栈顺序优化：先大后小
        if (left[1] - left[0]) > (right[1] - right[0]):
            if left[0] < left[1]: stack.append(left)
            if right[0] < right[1]: stack.append(right)
        else:
            if right[0] < right[1]: stack.append(right)
            if left[0] < left[1]: stack.append(left)

    return a
```

### 4) 三路快排（Dutch Flag）+ 栈实现

三路 partition 会得到两段边界 `[lo..lt-1]` 和 `[gt+1..hi]`，中间 `[lt..gt]` 都等于 pivot。

```python
def quicksort_3way_stack(a: List[int]) -> List[int]:
    n = len(a)
    # 长度小于 2 不需要排序
    if n < 2:
        return a

    # 栈保存待处理区间
    stack: List[Tuple[int, int]] = [(0, n - 1)]
    while stack:
        lo, hi = stack.pop()
        if lo >= hi:
            continue

        pivot = a[lo]  # 基准值
        lt = lo        # a[lo..lt-1] < pivot
        i = lo         # 当前扫描指针
        gt = hi        # a[gt+1..hi] > pivot

        while i <= gt:
            if a[i] < pivot:
                # 把小于 pivot 的元素换到左边，扩展 < 区
                a[lt], a[i] = a[i], a[lt]
                lt += 1
                i += 1
            elif a[i] > pivot:
                # 把大于 pivot 的元素换到右边，扩展 > 区（i 不前进）
                a[i], a[gt] = a[gt], a[i]
                gt -= 1
            else:
                # 等于 pivot，留在中间区，继续扫描
                i += 1

        # 中间 [lt..gt] 已经有序（都等于 pivot），仅处理两侧
        left = (lo, lt - 1)
        right = (gt + 1, hi)

        # 先压较大区间，减少栈峰值
        if (left[1] - left[0]) > (right[1] - right[0]):
            if left[0] < left[1]: stack.append(left)
            if right[0] < right[1]: stack.append(right)
        else:
            if right[0] < right[1]: stack.append(right)
            if left[0] < left[1]: stack.append(left)

    return a
```

### 5) 随机 pivot（栈版）

思路：每次处理一个区间前，先随机选 pivot 位置并交换到 partition 期望的位置即可。

#### 5.1 Lomuto + 随机 pivot + 栈

```python
def quicksort_lomuto_random_stack(a: List[int]) -> List[int]:
    n = len(a)
    # 小数组直接返回
    if n < 2:
        return a

    # 栈模拟递归
    stack: List[Tuple[int, int]] = [(0, n - 1)]
    while stack:
        lo, hi = stack.pop()
        if lo >= hi:
            continue

        # 在当前区间随机选 pivot，下放到 hi 以复用 Lomuto 分区
        r = random.randint(lo, hi)
        a[r], a[hi] = a[hi], a[r]  # pivot 放到 hi
        p = partition_lomuto(a, lo, hi)

        left = (lo, p - 1)
        right = (p + 1, hi)

        # 先压大区间，后处理小区间
        if (left[1] - left[0]) > (right[1] - right[0]):
            if left[0] < left[1]: stack.append(left)
            if right[0] < right[1]: stack.append(right)
        else:
            if right[0] < right[1]: stack.append(right)
            if left[0] < left[1]: stack.append(left)

    return a
```

#### 5.2 Hoare + 随机 pivot + 栈

```python
def quicksort_hoare_random_stack(a: List[int]) -> List[int]:
    n = len(a)
    # 长度小于 2 时已经有序
    if n < 2:
        return a

    # 栈里存要处理的闭区间
    stack: List[Tuple[int, int]] = [(0, n - 1)]
    while stack:
        lo, hi = stack.pop()
        if lo >= hi:
            continue

        # 随机选 pivot，并放到 lo 位置以匹配 Hoare 分区实现
        r = random.randint(lo, hi)
        a[r], a[lo] = a[lo], a[r]  # pivot 放到 lo
        p = partition_hoare(a, lo, hi)

        left = (lo, p)
        right = (p + 1, hi)

        # 压栈顺序优化
        if (left[1] - left[0]) > (right[1] - right[0]):
            if left[0] < left[1]: stack.append(left)
            if right[0] < right[1]: stack.append(right)
        else:
            if right[0] < right[1]: stack.append(right)
            if left[0] < left[1]: stack.append(left)

    return a
```

### 6) 小数组切插入排序（工程优化）+ 栈版（以 Lomuto 为例）

```python
def quicksort_lomuto_insertion_stack(a: List[int], cutoff: int = 16) -> List[int]:
    n = len(a)
    # 小于 2 个元素不用排
    if n < 2:
        return a

    # 栈实现的快排主循环
    stack: List[Tuple[int, int]] = [(0, n - 1)]
    while stack:
        lo, hi = stack.pop()
        if lo >= hi:
            continue

        # 小区间切换为插入排序，减少快排常数开销
        if hi - lo + 1 <= cutoff:
            insertion_sort_range(a, lo, hi)
            continue

        # 大区间：随机 pivot + Lomuto 分区
        r = random.randint(lo, hi)
        a[r], a[hi] = a[hi], a[r]  # 也顺便随机 pivot
        p = partition_lomuto(a, lo, hi)

        left = (lo, p - 1)
        right = (p + 1, hi)

        # 先压大的区间，降低栈深
        if (left[1] - left[0]) > (right[1] - right[0]):
            if left[0] < left[1]: stack.append(left)
            if right[0] < right[1]: stack.append(right)
        else:
            if right[0] < right[1]: stack.append(right)
            if left[0] < left[1]: stack.append(left)

    return a
```

### 7) 一键自测（可选）

```python
def is_sorted(a: List[int]) -> bool:
    # 判断数组是否单调非降
    return all(a[i] <= a[i+1] for i in range(len(a)-1))

def _demo():
    # 待测试的各版本快排函数
    funcs = [
        quicksort_lomuto_stack,
        quicksort_hoare_stack,
        quicksort_hole_stack,
        quicksort_3way_stack,
        quicksort_lomuto_random_stack,
        quicksort_hoare_random_stack,
        # 这里用 lambda 固定 cutoff 参数
        lambda x: quicksort_lomuto_insertion_stack(x, cutoff=16),
    ]

    # 覆盖：空数组、单元素、顺序、逆序、重复元素、随机数组
    tests = [
        [],
        [1],
        [2, 1],
        [1, 2, 3, 4, 5],
        [5, 4, 3, 2, 1],
        [3, 1, 2, 3, 3, 0, 3, 2],
        [random.randint(0, 10) for _ in range(200)],
    ]

    for f in funcs:
        for t in tests:
            # 每次复制原测试数据，避免被上一次排序污染
            a = t.copy()
            f(a)  # 原地排序
            # 排序失败时抛出函数名、原数据和结果，便于定位
            assert is_sorted(a), (f.__name__, t, a)
    print("all ok")

if __name__ == "__main__":
    # 直接运行本文件时执行自测
    _demo()
```

### 0) 工具：校验用

```python
from __future__ import annotations
from dataclasses import dataclass
from typing import List, Callable, Tuple
import random

def is_sorted(a: List[int]) -> bool:
    return all(a[i] <= a[i+1] for i in range(len(a)-1))
```



### 1) Lomuto 分区（pivot 取右端）

```python
from typing import List

def partition_lomuto(a: List[int], lo: int, hi: int) -> int:
    pivot = a[hi]
    i = lo - 1
    for j in range(lo, hi):
        if a[j] < pivot:
            i += 1
            a[i], a[j] = a[j], a[i]
    a[i+1], a[hi] = a[hi], a[i+1]
    return i + 1

def quicksort_lomuto(a: List[int]) -> List[int]:
    def _qs(lo: int, hi: int) -> None:
        if lo >= hi:
            return
        p = partition_lomuto(a, lo, hi)
        _qs(lo, p - 1)
        _qs(p + 1, hi)
    _qs(0, len(a) - 1)
    return a
```



### 2) Hoare 分区（经典双指针，pivot 常取左端）

Hoare 返回的是“分割点 j”，递归边界是 `[lo..j]` 和 `[j+1..hi]`（注意和 Lomuto 不一样）。

```python
def partition_hoare(a: List[int], lo: int, hi: int) -> int:
    pivot = a[lo]
    i = lo - 1
    j = hi + 1
    while True:
        i += 1
        while a[i] < pivot:
            i += 1
        j -= 1
        while a[j] > pivot:
            j -= 1
        if i >= j:
            return j
        a[i], a[j] = a[j], a[i]

def quicksort_hoare(a: List[int]) -> List[int]:
    def _qs(lo: int, hi: int) -> None:
        if lo >= hi:
            return
        p = partition_hoare(a, lo, hi)
        _qs(lo, p)
        _qs(p + 1, hi)
    _qs(0, len(a) - 1)
    return a
```

### 3) 挖坑法（hole method，面试常见）

```python
def partition_hole(a: List[int], lo: int, hi: int) -> int:
    pivot = a[lo]
    i, j = lo, hi
    while i < j:
        while i < j and a[j] >= pivot:
            j -= 1
        a[i] = a[j]  # 右边找到小的，填到左坑

        while i < j and a[i] <= pivot:
            i += 1
        a[j] = a[i]  # 左边找到大的，填到右坑
    a[i] = pivot
    return i

def quicksort_hole(a: List[int]) -> List[int]:
    def _qs(lo: int, hi: int) -> None:
        if lo >= hi:
            return
        p = partition_hole(a, lo, hi)
        _qs(lo, p - 1)
        _qs(p + 1, hi)
    _qs(0, len(a) - 1)
    return a
```

### 4) 三路快排（Dutch National Flag，重复元素多最强）

```python
def quicksort_3way(a: List[int]) -> List[int]:
    def _qs(lo: int, hi: int) -> None:
        if lo >= hi:
            return
        pivot = a[lo]
        lt = lo       # a[lo..lt-1] < pivot
        i = lo        # a[lt..i-1] == pivot
        gt = hi       # a[gt+1..hi] > pivot
        while i <= gt:
            if a[i] < pivot:
                a[lt], a[i] = a[i], a[lt]
                lt += 1
                i += 1
            elif a[i] > pivot:
                a[i], a[gt] = a[gt], a[i]
                gt -= 1
            else:
                i += 1
        _qs(lo, lt - 1)
        _qs(gt + 1, hi)
    _qs(0, len(a) - 1)
    return a
```

### 5) 随机 pivot（防止已排序退化）

这里给一个通用包装：对任意 partition（需要 pivot 在 `lo` 或 `hi`）做随机化。

* 对 **Lomuto**：随机一个位置与 `hi` 交换，让 pivot 在末尾
* 对 **Hoare / 挖坑**：随机一个位置与 `lo` 交换，让 pivot 在开头

```python
def quicksort_lomuto_random(a: List[int]) -> List[int]:
    def _qs(lo: int, hi: int) -> None:
        if lo >= hi:
            return
        r = random.randint(lo, hi)
        a[r], a[hi] = a[hi], a[r]
        p = partition_lomuto(a, lo, hi)
        _qs(lo, p - 1)
        _qs(p + 1, hi)
    _qs(0, len(a) - 1)
    return a

def quicksort_hoare_random(a: List[int]) -> List[int]:
    def _qs(lo: int, hi: int) -> None:
        if lo >= hi:
            return
        r = random.randint(lo, hi)
        a[r], a[lo] = a[lo], a[r]
        p = partition_hoare(a, lo, hi)
        _qs(lo, p)
        _qs(p + 1, hi)
    _qs(0, len(a) - 1)
    return a
```

### 6) 非递归快排（用栈模拟，示例用 Lomuto）

```python
def quicksort_iterative_lomuto(a: List[int]) -> List[int]:
    if len(a) < 2:
        return a
    stack: List[Tuple[int, int]] = [(0, len(a) - 1)]
    while stack:
        lo, hi = stack.pop()
        if lo >= hi:
            continue
        p = partition_lomuto(a, lo, hi)
        # 为了更省栈：先压大的区间，后处理小的区间（让栈深更小）
        left = (lo, p - 1)
        right = (p + 1, hi)
        if (left[1] - left[0]) > (right[1] - right[0]):
            stack.append(left)
            stack.append(right)
        else:
            stack.append(right)
            stack.append(left)
    return a
```

### 7) 小数组切换插入排序（工程常用优化）

```python
def insertion_sort_range(a: List[int], lo: int, hi: int) -> None:
    for i in range(lo + 1, hi + 1):
        key = a[i]
        j = i - 1
        while j >= lo and a[j] > key:
            a[j + 1] = a[j]
            j -= 1
        a[j + 1] = key

def quicksort_lomuto_with_insertion(a: List[int], cutoff: int = 16) -> List[int]:
    def _qs(lo: int, hi: int) -> None:
        if hi - lo + 1 <= cutoff:
            insertion_sort_range(a, lo, hi)
            return
        r = random.randint(lo, hi)
        a[r], a[hi] = a[hi], a[r]
        p = partition_lomuto(a, lo, hi)
        _qs(lo, p - 1)
        _qs(p + 1, hi)
    if len(a) > 1:
        _qs(0, len(a) - 1)
    return a
```
